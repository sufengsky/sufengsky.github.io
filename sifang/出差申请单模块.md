  sdate: "2010-10-11",
  edate: "2010-10-11",
};

  (d1.sdate == d1.edate && d2.sdate == d2.edate && d1.edate == d2.sdate) ||

    this.moment(d1.sdate).isAfter(d2.sdate));
console.log(flag);
